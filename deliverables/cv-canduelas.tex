\section{Artistic Practice}

\entry{Warmipa Kunkan Art}{Cardiff, Wales, UK}{Independent Visual Artist}{2023 -- Present}
\begin{itemize}
  \item Exhibited in 20+ juried and open exhibitions across Wales, Italy, and Peru over a 30-year artistic career
  \item Created multidisciplinary work spanning painting, sculpture, weaving, encaustic, and cyanotype, selling through Saatchi Art and private commissions
  \item Produced socially engaged installations addressing forced sterilisations in Peru, exhibited at Turner House, G39, and Made in Roath
\end{itemize}
\vspace{0.2em}

\entry{Llanover Hall Art Centre}{Cardiff, Wales, UK}{Artist-in-Residence}{2023 -- 2024}
\begin{itemize}
  \item Led experimental workshops in drawing, pottery, and painting, engaging community members across all ages
\end{itemize}
\vspace{0.2em}

\entry{MilaSelsh Art Ceramic Studio}{Lima, Peru}{Founder \& Instructor}{2004 -- 2009}
\begin{itemize}
  \item Founded ceramic workshop teaching casting, glazing, firing, and clay modelling to children ages 7--12
\end{itemize}
\vspace{0.2em}

\entry{Independent Fashion Designer}{Lima \& Pucallpa, Peru}{Designer \& Workshop Owner}{1992 -- 2001}
\begin{itemize}
  \item Operated custom clothing manufacturing workshop for 9 years, earning Recognition and Honour to Merit at the 1st Regional Contest of Amazonian Crafts (1997)
\end{itemize}
